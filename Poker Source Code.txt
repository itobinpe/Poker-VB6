Private Sub cmdCall_Click()

cmdCall.Visible = False
cmdRaise.Visible = False
cmdFold.Visible = False

If PlayerBal > CallBal Then

    Pot = Pot + CallBal
    PlayerBal = PlayerBal - CallBal
    lblPot.Caption = "Pot: $" & Pot
    lblPlayerBal = "Balance: $" & PlayerBal
    lblPM.Visible = True
    lblPM.Caption = "You have called."
    PlayerTurn = False
    PlayerTime = 2
    TurnTaken = True

ElseIf PlayerBal <= CallBal Then 'not enough for call, all in

    CallBal = PlayerBal
    Pot = Pot + CallBal
    PlayerBal = PlayerBal - CallBal
    lblPot.Caption = "Pot: $" & Pot
    lblPlayerBal = "Balance: $" & PlayerBal
    lblPM.Visible = True
    lblPM.Caption = "You went all in."
    PlayerTurn = False
    PlayerTime = 2
    TurnTaken = True
    PlayerAllIn = True

ElseIf PlayerSmallBlind = True And CallBal <> SmallBlind * 2 Then
            
    tempSBal = CallBal - SmallBlind
    PlayerBal = PlayerBal - tempSBal
    Pot = Pot + CallBal
    lblPlayerBal.Caption = "Balance: $" & PlayerBal
    lblPM.Caption = "You have called."
    lblPot.Caption = "Pot: $" & Pot
    lblPM.Visible = True
    PlayerTurn = False

End If

End Sub

Public Sub CheckFoldP()

If PlayerFold = True And Bot1Fold = True And Bot2Fold = True And Bot3Fold = True Then 'add Pot Bal to winner and restart cards (game)

    Bot4Win = True
    MsgBox "Everyone else has folded. Bot 4 wins.", , "Notice"

ElseIf PlayerFold = True And Bot1Fold = True And Bot2Fold = True And Bot4Fold = True Then

    Bot3Win = True
    MsgBox "Everyone else has folded. Bot 3 wins.", , "Notice"

ElseIf PlayerFold = True And Bot1Fold = True And Bot3Fold = True And Bot4Fold = True Then

    Bot2Win = True
    MsgBox "Everyone else has folded. Bot 2 wins.", , "Notice"
    
ElseIf PlayerFold = True And Bot2Fold = True And Bot3Fold = True And Bot4Fold = True Then

    Bot1Win = True
    MsgBox "Everyone else has folded. Bot 1 wins.", , "Notice"
    
ElseIf Bot1Fold = True And Bot2Fold = True And Bot3Fold = True And Bot4Fold = True Then

    PlayerWin = True
    MsgBox "Everyone else has folded. Player wins.", , "Notice"

Else

    Call TurnP

End If

End Sub
